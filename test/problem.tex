\section{Problem Formulation}

\subsection{Decision Variables}
Let $x_i \in [0,1]$ be the normalized price variable for platform $i$, where $i = 1, 2, \ldots, n$ platforms.

The actual price is defined as:
\begin{equation}
	p_i(x_i) = p_{\min} + x_i(p_{\max} - p_{\min})
\end{equation}

This normalization allows optimization over the unit interval while mapping to actual prices within the global bounds.

\subsection{Given Parameters}
\begin{itemize}
	\item $r_i$: Royalty rate for platform $i$ (as decimal, e.g., 0.70 for 70\%)
	\item $m_i$: Market share/relative demand for platform $i$ (normalized so $\sum_{i=1}^n m_i = 1$)
	\item $\bar{p}$: Average price from market data analysis
	\item $\lambda_v$: Weight parameter for price variance penalty
	\item $\lambda_d$: Weight parameter for deviation from market average penalty
	\item $\lambda_c$: Weight parameter for competitive advantage penalty
	\item $\alpha$: Discount factor for your store (e.g., 0.85 for 15\% discount advantage)
\end{itemize}

\subsection{Enhanced Objective Function}

The multi-objective optimization problem becomes:
\begin{equation}
	\max_{x \in [0,1]^n} \left[ \text{Revenue} - \lambda_v \cdot \text{Variance} - \lambda_d \cdot \text{Deviation} - \lambda_c \cdot \text{Competitive Penalty} \right]
\end{equation}

Where:

\textbf{Revenue Term:}
\begin{equation}
	\text{Revenue} = \sum_{i=1}^n r_i \cdot p_i(x_i) \cdot m_i \cdot f_i(x_i)
\end{equation}

\textbf{Price Variance Penalty:}
\begin{equation}
	\text{Variance} = \frac{1}{n} \sum_{i=1}^n (p_i(x_i) - \bar{p}_{\text{our}})^2
\end{equation}
where $\bar{p}_{\text{our}} = \frac{1}{n}\sum_{i=1}^n p_i(x_i)$

\textbf{Market Deviation Penalty:}
\begin{equation}
	\text{Deviation} = \left(\frac{1}{n}\sum_{i=1}^n p_i(x_i) - \bar{p}\right)^2
\end{equation}

\textbf{Competitive Advantage Penalty:}
\begin{equation}
	\text{Competitive Penalty} = \max(0, p_1(x_1) - \alpha \cdot \min_{j \neq 1} p_j(x_j))^2
\end{equation}

\subsection{Demand Function $f_i(x_i)$}
To model price sensitivity, we use an exponential demand function:
\begin{equation}
	f_i(x_i) = e^{-\beta_i (p_i(x_i) - p_{\text{ref}})}
\end{equation}

Where:
\begin{itemize}
	\item $\beta_i > 0$ is the price sensitivity parameter for platform $i$
	\item $p_{\text{ref}}$ is a reference price (could be $\bar{p}$ or platform-specific)
\end{itemize}

\section{Constraints}

\subsection{Essential Bounds}
\begin{equation}
	0 \leq x_i \leq 1 \quad \forall i
\end{equation}

\subsection{Optional Monotonicity (from original paper)}
If you want to preserve format hierarchy:
\begin{equation}
	x_{i+1} - x_i \geq \delta \quad \text{for ordered formats}
\end{equation}

Only use this if platforms have clear value ordering (e.g., hardcover $>$ paperback $>$ ebook).

\subsection{Maximum Price Spread (Optional)}
To prevent extreme price differences:
\begin{equation}
	\max_i p_i(x_i) - \min_i p_i(x_i) \leq \text{spread}_{\max}
\end{equation}

\section{Complete Mathematical Model}

\subsection{Enhanced Objective with Soft Constraints}
Convert to minimization problem for standard solvers:

\begin{align}
	\min_{x} \Bigg[ & -\sum_{i=1}^n r_i \cdot p_i(x_i) \cdot m_i \cdot e^{-\beta_i (p_i(x_i) - \bar{p})} \nonumber \\
	                & + \lambda_v \sum_{i=1}^n (p_i(x_i) - \bar{p}_{\text{our}})^2 \nonumber                       \\
	                & + \lambda_d \left(\bar{p}_{\text{our}} - \bar{p}\right)^2 \nonumber                          \\
	                & + \lambda_c \cdot \max(0, p_1(x_1) - \alpha \cdot \min_{j \neq 1} p_j(x_j))^2 \Bigg]
\end{align}

Subject to:
\begin{itemize}
	\item $0 \leq x_i \leq 1 \quad \forall i$ (essential bounds only)
	\item Optional monotonicity constraints if format hierarchy exists
	\item Optional maximum spread constraint if needed
\end{itemize}

where $p_i(x_i) = p_{\min} + x_i(p_{\max} - p_{\min})$

\section{Seasonal Discount Strategy}

For twice-yearly competitive advantages, modify the competitive penalty during discount periods:

\textbf{Regular periods:}
\begin{equation}
	\lambda_c \cdot \max(0, p_1(x_1) - \alpha \cdot \min_{j \neq 1} p_j(x_j))^2
\end{equation}

\textbf{Discount periods:}
\begin{equation}
	\lambda_c \cdot \max(0, p_1(x_1) - \alpha_{\text{discount}} \cdot \min_{j \neq 1} p_j(x_j))^2
\end{equation}

where $\alpha_{\text{discount}} < \alpha$ and $\lambda_c$ can be increased during discount periods to strengthen the competitive push.

\section{Implementation Notes}

\subsection{Parameter Selection}
\begin{itemize}
	\item $\lambda_v \in [0.1, 1.0]$: Controls price variance (higher = more uniform pricing)
	\item $\lambda_d \in [0.1, 0.5]$: Controls deviation from market average
	\item $\lambda_c \in [0.1, 2.0]$: Controls competitive advantage strength
	\item $\beta_i$: Platform-specific elasticity (estimate from historical data)
	\item $\alpha \in [0.8, 0.95]$: Competitive discount (20\% to 5\% advantage)
\end{itemize}

\subsection{Solver Recommendations}
\begin{enumerate}
	\item \textbf{SLSQP} (Sequential Least Squares Programming)
	\item \textbf{Trust-Region Constrained}
	\item \textbf{Interior Point Methods}
\end{enumerate}

\subsection{Multi-Objective Handling}
Alternative approach using weighted sum:
\begin{equation}
	\min_{x} \left[ w_1 \cdot (-\text{Revenue}) + w_2 \cdot \text{Variance} + w_3 \cdot \text{Deviation} + w_4 \cdot \text{Competitive Penalty} \right]
\end{equation}

Where $w_1 + w_2 + w_3 + w_4 = 1$ and weights reflect business priorities.
